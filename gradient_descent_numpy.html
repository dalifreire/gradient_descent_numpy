def funcao(x, func='sigmoid'):
    
    if func == 'reta':
        f_x = 3*x
    elif func == 'parabola':
        f_x = x**2
    elif func == 'seno':
        f_x = np.sin(x)
    elif func == 'relu':
        f_x = np.maximum(0, x)
    elif func == 'tanh':
        f_x = np.tanh(x)
    else:
        f_x = 1/(1 + np.exp(-x))
    
    return f_x

def derivada(x, func='sigmoid'):
    
    if func == 'reta':
        x_deriv = 3
    elif func == 'parabola':
        x_deriv = 2*x
    elif func == 'seno':
        x_deriv = np.cos(x)
    elif func == 'tanh':
        x_deriv = (1 - (np.tanh(x) ** 2))
    elif func == 'relu':
        x_deriv = np.where(x <= 0, 0, 1)
    else:
        x_deriv = (1/(1 + np.exp(-x))) * (1 - (1/(1 + np.exp(-x))))
    
    return x_deriv

def execute(x_new, epochs, lr, func='sigmoid'):
    
    x_prev = 0
    x_values, y_values = [x_new], [funcao(x_new, func)]
    
    cont = 0
    while abs(x_new - x_prev) > 0.001 and cont < epochs:
        
        x_prev = x_new
        
        d_x = derivada(x_prev, func)
        x_new = x_prev - (lr * d_x)
        
        x_values.append(x_new)
        y_values.append(funcao(x_new, func))
        
        cont+=1
        
    print (f"Mnimo local ocorreu em: x={str(x_new)}, y={str(funcao(x_new, func))} [{x_new - x_prev}]")
    print ("pocas executadas: " + str(cont))
    
    return x_values, y_values
